systemd CPU Pinning (titan-core.service)

[Unit]
Description=TITAN Core (C++ execution engine)
After=network-online.target redis.service
Requires=redis.service

[Service]
Type=simple
ExecStart=/opt/titan/bin/titan-core --config /etc/titan/titan-core.yaml
Restart=always
RestartSec=2

# CPU isolation
CPUAffinity=2,3
AllowedCPUs=2,3
NUMAPolicy=bind
NUMAMask=0

# Real-time scheduling
Nice=-11
IOSchedulingClass=realtime
IOSchedulingPriority=0

# Memory
MemoryAccounting=true
MemoryHigh=4G
MemoryMax=6G
MemorySwapMax=0

# I/O
IOWeight=1000
BlockIOWeight=1000

# Security hardening
NoNewPrivileges=true
ProtectSystem=strict
ProtectHome=true
PrivateTmp=true
ReadWritePaths=/var/lib/titan /var/log/titan

[Install]
WantedBy=multi-user.target
